import { Component } from '@angular/core';

@Component({
  selector: 'app-artworks',
  templateUrl: './artworks.component.html',
  styleUrls: ['./artworks.component.scss']
})
export class ArtworksComponent {
  filter: string[] = [];
  sort: string = '';

  filterChanged(selectedStyles: string[]): void {
    this.filter = selectedStyles;
  }

  sortChanged(sort: string): void {
    this.sort = sort;
  }
}
<div class="artworks-container">
    <!-- Filter and Sort Form -->
    <app-filter-sort-form (filterChange)="filterChanged($event)" (sortChange)="sortChanged($event)"></app-filter-sort-form>
  
    <!-- Artwork List -->
    <app-artwork-list [filter]="filter" [sort]="sort"></app-artwork-list>
  </div>
